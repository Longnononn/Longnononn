-- full code with Auto-Hop Full Moon button integrated
local GuiLibrary = {}

-- Giao diện
local ScreenGui = Instance.new("ScreenGui")
local Frame = Instance.new("Frame")
local AutoHopMoon = Instance.new("TextButton")

ScreenGui.Name = "longdzzGUI"
ScreenGui.Parent = game.CoreGui

Frame.Name = "MainFrame"
Frame.Parent = ScreenGui
Frame.BackgroundColor3 = Color3.fromRGB(30, 30, 30)
Frame.Position = UDim2.new(0.3, 0, 0.3, 0)
Frame.Size = UDim2.new(0, 200, 0, 100)
Frame.Active = true
Frame.Draggable = true

AutoHopMoon.Name = "AutoHopMoon"
AutoHopMoon.Parent = Frame
AutoHopMoon.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
AutoHopMoon.Position = UDim2.new(0.1, 0, 0.2, 0)
AutoHopMoon.Size = UDim2.new(0.8, 0, 0.3, 0)
AutoHopMoon.Font = Enum.Font.SourceSans
AutoHopMoon.Text = "Auto-Hop Full Moon"
AutoHopMoon.TextColor3 = Color3.fromRGB(255, 255, 255)
AutoHopMoon.TextSize = 14

-- Tích hợp lấy JobId từ bot Discord
AutoHopMoon.MouseButton1Click:Connect(function()
    local HttpService = game:GetService("HttpService")
    local success, response = pcall(function()
        return HttpService:GetAsync("http://localhost:8080/jobid")
    end)

    if success and response and string.match(response, "[%w-]+") then
        local TeleportService = game:GetService("TeleportService")
        TeleportService:TeleportToPlaceInstance(2753915549, response, game.Players.LocalPlayer)
    else
        warn("Không thể lấy JobId mới nhất từ bot Discord.")
    end
end)
