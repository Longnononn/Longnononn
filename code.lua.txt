-- GUI Auto-Hop Full Moon từ Discord Bot + Replit Server
local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")

-- Giao diện
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "longdzzGUI"
ScreenGui.Parent = game.CoreGui

local Frame = Instance.new("Frame")
Frame.Name = "MainFrame"
Frame.Parent = ScreenGui
Frame.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
Frame.Position = UDim2.new(0.3, 0, 0.3, 0)
Frame.Size = UDim2.new(0, 240, 0, 120)
Frame.Active = true
Frame.Draggable = true
Frame.Visible = false

local UICorner1 = Instance.new("UICorner", Frame)
UICorner1.CornerRadius = UDim.new(0, 10)

local UIStroke = Instance.new("UIStroke", Frame)
UIStroke.Color = Color3.fromRGB(85, 85, 255)
UIStroke.Thickness = 1.5

local AutoHopMoon = Instance.new("TextButton")
AutoHopMoon.Name = "AutoHopMoon"
AutoHopMoon.Parent = Frame
AutoHopMoon.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
AutoHopMoon.Position = UDim2.new(0.1, 0, 0.35, 0)
AutoHopMoon.Size = UDim2.new(0.8, 0, 0.3, 0)
AutoHopMoon.Font = Enum.Font.Gotham
AutoHopMoon.Text = "Auto-Hop Full Moon"
AutoHopMoon.TextColor3 = Color3.fromRGB(255, 255, 255)
AutoHopMoon.TextSize = 16
Instance.new("UICorner", AutoHopMoon).CornerRadius = UDim.new(0, 8)

-- Hover
AutoHopMoon.MouseEnter:Connect(function()
    AutoHopMoon.BackgroundColor3 = Color3.fromRGB(65, 65, 65)
end)
AutoHopMoon.MouseLeave:Connect(function()
    AutoHopMoon.BackgroundColor3 = Color3.fromRGB(45, 45, 45)
end)

-- Nút mở/ẩn GUI
local ToggleButton = Instance.new("ImageButton")
ToggleButton.Name = "ToggleButton"
ToggleButton.Parent = ScreenGui
ToggleButton.BackgroundColor3 = Color3.fromRGB(85, 85, 255)
ToggleButton.Position = UDim2.new(0.05, 0, 0.2, 0)
ToggleButton.Size = UDim2.new(0, 50, 0, 50)
ToggleButton.Image = "rbxassetid://17250817867"
ToggleButton.Active = true
ToggleButton.Draggable = true
Instance.new("UICorner", ToggleButton).CornerRadius = UDim.new(1, 0)

ToggleButton.MouseButton1Click:Connect(function()
    Frame.Visible = not Frame.Visible
end)

-- Nút "Auto-Hop" sẽ lấy JobId từ Replit Server
AutoHopMoon.MouseButton1Click:Connect(function()
    local success, response = pcall(function()
        -- Thay link dưới bằng REPLIT thật của bạn
        return HttpService:GetAsync("https://your-repl-name.repl.co/jobid")
    end)

    if success and response and string.match(response, "[%w-]+") then
        TeleportService:TeleportToPlaceInstance(2753915549, response, game.Players.LocalPlayer)
    else
        warn("❌ Không thể lấy JobId mới nhất từ bot Discord.")
    end
end)
