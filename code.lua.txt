RunService.Stepped:Connect(function()
	if noclipEnabled and character then
		for _, part in ipairs(character:GetDescendants()) do
			if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
				part.CanCollide = false
			end
		end
	end
end)

local function performQuickSteal()
	if not hrp then return end
	local remote = ReplicatedStorage:FindFirstChild("StealPet", true)
	if not remote or not remote:IsA("RemoteEvent") then return end

	local origin = hrp.CFrame
	local pos = hrp.Position

	for _, obj in pairs(Workspace:GetDescendants()) do
		if obj.Name == "StealableItem" then
			local part = obj:IsA("BasePart") and obj or obj:FindFirstChildWhichIsA("BasePart")
			if part and (part.Position - pos).Magnitude <= stealRange then
				hrp.CFrame = CFrame.new(part.Position + Vector3.new(0, 1, 0))
				task.wait(0.03)
				remote:FireServer(obj)
				task.wait(0.05)
				hrp.CFrame = origin
				task.wait(0.05)
			end
		end
	end
end

task.spawn(function()
	while true do
		if quickStealEnabled and tick() - lastStealAttemptTime >= stealActionCooldown then
			performQuickSteal()
			lastStealAttemptTime = tick()
		end
		task.wait(0.1)
	end
end)

	task.wait(0.25)
	if speedEnabled then
		humanoid.WalkSpeed = walkSpeedBoost
	end
end)

-- Anti Kick
pcall(function()
	local mt = getrawmetatable(game)
	setreadonly(mt, false)
	local old = mt.__namecall
	mt.__namecall = newcclosure(function(self, ...)
		if getnamecallmethod() == "Kick" then return end
		return old(self, ...)
	end)
	setreadonly(mt, true)
end)

-- ESP
local function createESP(part)
	if not part:IsA("BasePart") then return end
	if part:FindFirstChild("ESP_HL") then return end

	local hl = Instance.new("Highlight")
	hl.Name = "ESP_HL"
	hl.Adornee = part
	hl.FillTransparency = 1
	hl.OutlineTransparency = 0
	hl.OutlineColor = Color3.fromRGB(0, 255, 0)
	hl.Parent = part
end

local function findBasePart(obj)
	if obj:IsA("BasePart") then
		return obj
	elseif obj:IsA("Model") or obj:IsA("Folder") then
		for _, v in ipairs(obj:GetDescendants()) do
			if v:IsA("BasePart") then return v end
		end
	end
end

local function applyESPToAll()
	for _, obj in ipairs(Workspace:GetDescendants()) do
		if obj.Name == "StealableItem" then
			local bp = findBasePart(obj)
			if bp then createESP(bp) end
		end
	end
end

Workspace.DescendantAdded:Connect(function(desc)
	if desc.Name == "StealableItem" then
		task.wait(0.1)
		local bp = findBasePart(desc)
		if bp then createESP(bp) end
	end
end)

applyESPToAll()
